/**
 * Initialize projects functionality
 */
function initProjects() {
    const projectGrid = document.querySelector('.project-grid');
    const filterButtons = document.querySelectorAll('.project-filters .filter-btn');

    let currentFilter = 'all';

    // Generate initial projects
    renderProjects();

    // Filter button click handler
    filterButtons.forEach(button => {
        button.addEventListener('click', function(e) {
            // Prevent default action and scrolling
            e.preventDefault();
            e.stopPropagation();
            
            // Store current scroll position
            const currentScrollY = window.scrollY;
            
            // Update active button - only within main projects section
            filterButtons.forEach(btn => btn.classList.remove('active'));
            this.classList.add('active');

            // Update filter
            currentFilter = this.getAttribute('data-filter');

            // Re-render projects
            renderProjects();
            
            // Restore scroll position to prevent jumping
            window.scrollTo(0, currentScrollY);
        });
    });

    /**
     * Render projects based on current filter - show all projects except personal
     */
    function renderProjects() {
        // Clear project grid
        projectGrid.innerHTML = '';

        // Filter projects - exclude personal projects from main section
        let filteredProjects = currentFilter === 'all'
            ? projects.filter(project => project.category !== 'personal')
            : projects.filter(project => project.category === currentFilter && project.category !== 'personal');

        // If "All" filter is selected, move "Company" projects to the end
        if (currentFilter === 'all') {
            const companyProjects = filteredProjects.filter(project => project.category === 'company');
            const nonCompanyProjects = filteredProjects.filter(project => project.category !== 'company');
            filteredProjects = nonCompanyProjects.concat(companyProjects);
        }

        // Show all filtered projects
        filteredProjects.forEach((project, index) => {
            const projectCard = document.createElement('div');
            projectCard.className = 'project-card fade-in';

            const techTags = project.technologies.map(tech =>
                `<span class="project-tag">${tech}</span>`
            ).join('');

            const newProjectBadge = project.isNew ? '<div class="new-project-badge">New Project</div>' : '';

            let statusBadgeClass = project.status.toLowerCase().replace(' ', '-');
            if(statusBadgeClass === 'not-active'){
                statusBadgeClass = 'disabled';
            }

            projectCard.innerHTML = `
                <div class="project-image">
                    ${newProjectBadge}
                    <img src="${project.image}" alt="${project.title}">
                    <div class="project-overlay">
                        <h3>${project.title}</h3>
                    </div>
                </div>
                <div class="project-content">
                    <p class="project-description">${project.description}</p>
                    <div class="project-tags">
                        ${techTags}
                    </div>
                    <div class="project-metadata">
                        <div class="metadata-item">
                            <span class="metadata-label">${project.category === 'company' ? 'Company:' : project.category === 'personal' ? 'Type:' : 'Client:'}</span>
                            <span class="metadata-value">${project.client}</span>
                        </div>
                        <div class="metadata-item">
                            <span class="metadata-label">Site Status:</span>
                            <span class="metadata-value">
                                <span class="status-badge ${statusBadgeClass}">${project.status}</span>
                            </span>
                        </div>
                        <div class="metadata-item">
                            <span class="metadata-label">Last Update:</span>
                            <span class="metadata-value">${project.lastUpdated}</span>
                        </div>
                        <div class="metadata-item">
                            <span class="metadata-label">Maintenance:</span>
                            <span class="metadata-value">
                                <span class="maintenance-badge ${getMaintenanceBadgeClass(project.maintenanceMode)}">${project.maintenanceMode}</span>
                            </span>
                        </div>
                        <div class="metadata-item">
                            <span class="metadata-label">Project Type:</span>
                            <span class="metadata-value">${project.projectType}</span>
                        </div>
                        ${project.developmentPhase ? `
                        <div class="metadata-item">
                            <span class="metadata-label">Development:</span>
                            <span class="metadata-value">
                                <span class="development-phase-badge">${project.developmentPhase}</span>
                            </span>
                        </div>
                        ` : ''}
                    </div>
                    <div class="project-buttons">
                        <button class="project-link" onclick="openProjectModal('${project.title.replace(/'/g, "\\'")}')">
                            Learn More <i class="fas fa-arrow-right"></i>
                        </button>
                        <a href="${project.url}" target="_blank" rel="noopener noreferrer" class="project-link view-project-btn">
                            View Project <i class="fas fa-external-link-alt"></i>
                        </a>
                    </div>
                </div>
            `;

            projectGrid.appendChild(projectCard);
        });

        // Trigger animation for newly added elements
        setTimeout(() => {
            document.querySelectorAll('.project-card.fade-in').forEach(card => {
                card.classList.add('active');
            });
        }, 100);
    }
}

/**
 * Open project details modal
 */
function openProjectModal(projectTitle) {
    const project = projects.find(p => p.title === projectTitle);
    const modal = document.getElementById('projectModal');

    if (!project || !modal) {
        console.error('Project or modal not found');
        return;
    }

    // Populate modal content
    const modalTitle = document.getElementById('modalTitle');
    const modalStatus = document.getElementById('modalStatus');
    
    if (modalTitle) modalTitle.textContent = project.title;
    if (modalStatus) modalStatus.textContent = project.status;

    let modalStatusClass = project.status.toLowerCase().replace(' ', '-');
    if(modalStatusClass === 'not-active'){
        modalStatusClass = 'disabled';
    }

    if (modalStatus) modalStatus.className = `status-badge ${modalStatusClass}`;
    
    const modalDuration = document.getElementById('modalDuration');
    const modalClient = document.getElementById('modalClient');
    const modalLastUpdate = document.getElementById('modalLastUpdate');
    const maintenanceElement = document.getElementById('modalMaintenance');
    const modalProjectType = document.getElementById('modalProjectType');
    const modalDescription = document.getElementById('modalDescription');
    
    if (modalDuration) modalDuration.textContent = project.duration;
    if (modalClient) modalClient.textContent = project.client;
    if (modalLastUpdate) modalLastUpdate.textContent = project.lastUpdated;
    if (modalProjectType) modalProjectType.textContent = project.projectType;
    if (modalDescription) modalDescription.textContent = project.detailedDescription;
    
    if (maintenanceElement) {
        maintenanceElement.innerHTML = `<span class="maintenance-status ${getMaintenanceBadgeClass(project.maintenanceMode)}">${project.maintenanceMode}</span>`;
    }

    // Update the label in modal based on project category
    const clientLabelElement = document.querySelector('.modal-sidebar p:nth-child(6)');
    if (clientLabelElement && project.category === 'company') {
        clientLabelElement.innerHTML = `<strong>Company:</strong> <span id="modalClient">${project.client}</span>`;
    } else if (clientLabelElement && project.category === 'personal') {
        clientLabelElement.innerHTML = `<strong>Type:</strong> <span id="modalClient">${project.client}</span>`;
    } else if (clientLabelElement) {
        clientLabelElement.innerHTML = `<strong>Client:</strong> <span id="modalClient">${project.client}</span>`;
    }

    // Handle maintenance field - hide for personal projects
    if (maintenanceElement) {
        if (project.category === 'personal') {
            // Hide maintenance field for personal projects
            const maintenanceParent = maintenanceElement.closest('p');
            if (maintenanceParent) {
                maintenanceParent.style.display = 'none';
            }
        } else {
            // Show maintenance field for other projects
            const maintenanceParent = maintenanceElement.closest('p');
            if (maintenanceParent) {
                maintenanceParent.style.display = 'block';
            }
            maintenanceElement.innerHTML = `<span class="maintenance-status ${getMaintenanceBadgeClass(project.maintenanceMode)}">${project.maintenanceMode}</span>`;
        }
    }

    // Handle development phase display in modal
    const modalSidebar = document.querySelector('.modal-sidebar');
    let existingDevelopmentPhase = modalSidebar.querySelector('.development-phase-display');
    if (existingDevelopmentPhase) {
        existingDevelopmentPhase.remove();
    }

    if (project.developmentPhase) {
        const developmentPhaseElement = document.createElement('p');
        developmentPhaseElement.className = 'development-phase-display';
        developmentPhaseElement.innerHTML = `<strong>Development:</strong> <span class="development-phase-badge">${project.developmentPhase}</span>`;
        modalSidebar.insertBefore(developmentPhaseElement, modalSidebar.querySelector('.modal-technologies'));
    }
    
    const modalImage = document.getElementById('modalImage');
    const modalProjectLink = document.getElementById('modalProjectLink');
    
    if (modalImage) {
        modalImage.src = project.image;
        modalImage.alt = project.title;
    }
    if (modalProjectLink) {
        modalProjectLink.href = project.url;
    }

    // Populate technologies
    const techContainer = document.getElementById('modalTechnologies');
    if (techContainer) {
        techContainer.innerHTML = project.technologies.map(tech => 
            `<span class="tech-tag">${tech}</span>`
        ).join('');
    }

    // Show additional links if they exist
    const adminLink = document.getElementById('modalAdminLink');
    const candidateDashboardLink = document.getElementById('modalCandidateDashboardLink');

    if (project.adminLink) {
        adminLink.href = project.adminLink;
        adminLink.style.display = 'inline-block';
    } else {
        adminLink.style.display = 'none';
    }

    if (project.candidateDashboardLink) {
        candidateDashboardLink.href = project.candidateDashboardLink;
        candidateDashboardLink.style.display = 'inline-block';
    } else {
        candidateDashboardLink.style.display = 'none';
    }

    // Show modal
    modal.style.display = 'flex';
    document.body.style.overflow = 'hidden';
}

/**
 * Get maintenance badge CSS class based on status
 */
function getMaintenanceBadgeClass(maintenanceMode) {
    const status = maintenanceMode.toLowerCase();

    if (status.includes('open') || status.includes('active')) {
        return 'maintenance-open';
    } else if (status.includes('closed')) {
        return 'maintenance-closed';
    } else if (status.includes('none')) {
        return 'maintenance-none';
    } else {
        return 'maintenance-none';
    }
}

/**
 * Close project details modal
 */
function closeProjectModal() {
    const modal = document.getElementById('projectModal');
    modal.style.display = 'none';
    document.body.style.overflow = 'auto';
}

// Close modal on escape key
document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape') {
        closeProjectModal();
    }
});
